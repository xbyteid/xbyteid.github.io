File Management with pathlib

The pathlib module is the modern way to handle file operations:

from pathlib import Path
import shutil

# Find all PDFs recursively
pdfs = list(Path("documents").rglob("*.pdf"))

# Organize files by extension
for f in Path("downloads").iterdir():
    if f.is_file():
        dest = Path("organized") / f.suffix[1:]
        dest.mkdir(parents=True, exist_ok=True)
        shutil.move(str(f), str(dest / f.name))

# Batch rename with pattern
for i, f in enumerate(sorted(Path("photos").glob("*.jpg"))):
    f.rename(f.parent / f"photo_{i:04d}.jpg")

Web Scraping with requests + BeautifulSoup

For static pages, requests + BeautifulSoup is fast and simple:

import requests
from bs4 import BeautifulSoup

def scrape_articles(url):
    resp = requests.get(url, headers={"User-Agent": "Mozilla/5.0"})
    soup = BeautifulSoup(resp.text, "html.parser")

    articles = []
    for item in soup.select("article.post"):
        title = item.select_one("h2").get_text(strip=True)
        link = item.select_one("a")["href"]
        date = item.select_one("time")["datetime"]
        articles.append({"title": title, "url": link, "date": date})

    return articles

API Integration

Automate API calls with proper error handling and retries:

import requests
from time import sleep

class APIClient:
    def __init__(self, base_url, api_key):
        self.base_url = base_url
        self.session = requests.Session()
        self.session.headers["Authorization"] = f"Bearer {api_key}"

    def get(self, endpoint, retries=3):
        for attempt in range(retries):
            try:
                resp = self.session.get(f"{self.base_url}{endpoint}")
                resp.raise_for_status()
                return resp.json()
            except requests.RequestException as e:
                if attempt == retries - 1:
                    raise
                sleep(2 ** attempt)  # exponential backoff

# Usage
api = APIClient("https://api.example.com", "your-key")
data = api.get("/v1/users")

Scheduled Tasks with APScheduler

Run jobs on a schedule without cron:

from apscheduler.schedulers.blocking import BlockingScheduler

scheduler = BlockingScheduler()

@scheduler.scheduled_job('cron', hour=9, minute=0)
def daily_report():
    data = fetch_daily_metrics()
    send_slack_notification(format_report(data))

@scheduler.scheduled_job('interval', minutes=30)
def check_health():
    status = ping_services()
    if any(s != 200 for s in status.values()):
        alert_on_call(status)

scheduler.start()

Email Automation with smtplib

Send emails programmatically for alerts and reports:

import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText

def send_email(to, subject, html_body):
    msg = MIMEMultipart("alternative")
    msg["From"] = "alerts@yourdomain.com"
    msg["To"] = to
    msg["Subject"] = subject
    msg.attach(MIMEText(html_body, "html"))

    with smtplib.SMTP("smtp.gmail.com", 587) as server:
        server.starttls()
        server.login("user@gmail.com", "app-password")
        server.send_message(msg)
Pro tip: Use python-dotenv to load API keys and credentials from .env files. Never hardcode secrets in scripts.

The beauty of Python automation is composability. Chain these scripts together — scrape data, process it, store results, send notifications — and you've built a fully automated workflow running on a $5 VPS.
